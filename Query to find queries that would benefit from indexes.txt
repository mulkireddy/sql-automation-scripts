-- Query to find queries that would benefit from indexes
SELECT 
    migs.avg_total_user_cost * migs.avg_user_impact * (migs.user_seeks + migs.user_scans) AS improvement_measure,
    'CREATE INDEX idx_' + REPLACE(REPLACE(mid.statement, '[', ''), ']', '') + '_'
    + CAST(mig.index_group_handle AS VARCHAR(10)) + ' ON ' + mid.statement 
    + ' (' + ISNULL(mid.equality_columns, '') + ') ' AS create_index_statement
FROM sys.dm_db_missing_index_groups mig
INNER JOIN sys.dm_db_missing_index_group_stats migs ON migs.group_handle = mig.index_group_handle
INNER JOIN sys.dm_db_missing_index_details mid ON mig.index_handle = mid.index_handle
ORDER BY improvement_measure DESC;
